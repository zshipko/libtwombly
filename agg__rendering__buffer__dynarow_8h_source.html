29  // Rendering buffer class with dynamic allocation of the rows.
30  // The rows are allocated as needed when requesting for span_ptr().
31  // The class automatically calculates min_x and max_x for each row.
32  // Generally it's more efficient to use this class as a temporary buffer
33  // for rendering a few lines and then to blend it with another buffer.
34  //

67  // Allocate and clear the buffer

69  void init(unsigned width, unsigned height, unsigned byte_width)
70  {
71  unsigned i;
72  for(i = 0; i < m_height; ++i)
73  {
74  pod_allocator<int8u>::deallocate((int8u*)m_rows[i].ptr, m_byte_width);
75  }
76  if(width && height)
77  {
78  m_width = width;
79  m_height = height;
80  m_byte_width = byte_width;
81  m_rows.resize(height);
82  memset(&m_rows[0], 0, sizeof(row_data) * height);
83  }
84  }

91  // The main function used for rendering. Returns pointer to the
92  // pre-allocated span. Memory for the row is allocated as needed.
93  //--------------------------------------------------------------------
94  int8u* row_ptr(int x, int y, unsigned len)
95  {
96  row_data* r = &m_rows[y];
97  int x2 = x + len - 1;
98  if(r->ptr)
99  {
100  if(x < r->x1) { r->x1 = x; }
101  if(x2 > r->x2) { r->x2 = x2; }
102  }
103  else
104  {
105  int8u* p = pod_allocator<int8u>::allocate(m_byte_width);
106  r->ptr = p;
107  r->x1 = x;
108  r->x2 = x2;
109  memset(p, 0, m_byte_width);
110  }
111  return (int8u*)r->ptr;
112  }

117  row_data row (int y) const { return m_rows[y]; }
